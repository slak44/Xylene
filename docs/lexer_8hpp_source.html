151  std::string str = "";
152  while (current() != '"') {
153  if (isEOF()) throw Error("SyntaxError", "String literal has unmatched quote", getCurrentLine());

156  skip(2); // Skip escape code, eg '\n', '\t'

201 inline Fixity Lexer::determineFixity(Fixity afterBinaryOrPrefix, Fixity afterIdentOrParen, Fixity otherCases) const {
202  if (getTokenCount() == 0) return otherCases;
203  Fixity type;
204  if (peekBehind().isOperator() && (peekBehind().hasArity(BINARY) || peekBehind().hasFixity(PREFIX))) {
205  // If the thing before this op is a binary op or a prefix unary, it means this must be a prefix unary
206  type = afterBinaryOrPrefix;
207  } else if (peekBehind().isTerminal() || peekBehind().type == C_PAREN_RIGHT || (peekBehind().isOperator() && peekBehind().hasFixity(POSTFIX))) {
208  // If the thing before was an identifier/paren (or the postfix op attached to an ident), then this must be a postfix unary or a binary infix
209  type = afterIdentOrParen;
210  } else {
211  type = otherCases;
212  }
213  return type;
214 }

231 inline Token Lexer::getNumberToken(int radix) {

236  if (current() == '.') {
237  if (isFloat) throw Error("SyntaxError", "Malformed float, multiple decimal points", getCurrentLine());

239  number += current();
240  skip(1);
241  } else if (isdigit(current())) {
242  number += current();
243  skip(1);
244  } else if (

246  (current() >= 'a' && current() < 'a' + radix - 10)
247  ) {
248  number += current();
249  skip(1);
250  } else {
251  break;
252  }
253  }
254  noIncrement();
255  if (number.back() == '.') throw Error("SyntaxError", "Malformed float, missing digits after decimal point", getCurrentLine());
